# Turn on the Rewrite Engine
RewriteEngine On

# Ensure the base path is correct for your subfolder
RewriteBase /aged/

# 1. EXTERNAL REDIRECT: If a user types "details.php", 
# change the URL in their browser to just "details"
RewriteCond %{THE_REQUEST} ^[A-Z]{3,}\s([^.]+)\.php [NC]
RewriteRule ^ %1 [R=301,L]

# 2. INTERNAL REWRITE: If a user requests "details", 
# secretly serve "details.php" so the page actually loads
RewriteCond %{REQUEST_FILENAME} !-d
RewriteCond %{REQUEST_FILENAME}.php -f
RewriteRule ^(.*?)/?$ $1.php [L]
